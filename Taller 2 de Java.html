3. Realizar un programa en el cual se solicite el nombre, apellidos y edad de la persona. Si la persona es mayor o igual a 18 años, entonce se deberá imprimir en pantalla [Nombre completo] usted es mayor de edad, por lo tanto puede entrar a la fiesta. Si la edad de la persona es menor que 18 años, entonces, deberá imprimirse el siguiente mensaje: [Nombre completo] usted es menor de edad, por lo tanto, no puede entrar a la fiesta, por favor devuélvase a su casa.


package pseint.taller.dos;

import java.util.Scanner;

/**
 *
 * @author Juan Osorio
 */
public class Tres {
    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Digite su nombre");
        String nombre = sc.nextLine();
        System.out.println("Digite su apellido");
        String apellido = sc.nextLine();
        System.out.println("Digite su edad");
        int edad = sc.nextInt();
        if (edad < 18) {
            System.out.println(nombre + " " + apellido + " usted es menor de edad, por lo tanto, no puede entrar a la fiesta, por favor devuélvase a su casa");
        } else {
            System.out.println(nombre + " " + apellido + " usted es mayor de edad, por lo tanto puede entrar a la fiesta");
        }
    }
}

4. La video tienda que presta sus servicios de alquiler de películas a los usuarios del barrio el Porvenir, requiere de una aplicación que permita registrar el alquiler de las películas que adquieren sus usuarios. Para cada usuario se debe permitir la opción de alquilar película, consultar películas disponibles y recibir película en la video tienda con la opción de realizar anotaciones sobre estas si se llegan a presentar daños u otra novedad sobre la película.


package pseint.taller.dos;

import javax.swing.JOptionPane;

/**
 *
 * @author Juan Osorio
 */
public class Cuatro {

    String nombrePelicula, pelicula, peliculaDos, observacion, observacionDos;
    boolean disponibilidadP1, disponibilidadP2;
    int opcion;

    public void agregarDatos() {
        pelicula = "Avatar";
        peliculaDos = "Terminator";
        disponibilidadP1 = true;
        disponibilidadP2 = false;
    }

    public void mostrar() {
        opcion = Integer.parseInt(JOptionPane.showInputDialog("Elija una opción:\n1: Alquilar Película\n2: Consultar Película\n3: Recibir Película"));

        switch (opcion) {
            case 1 -> {
                nombrePelicula = JOptionPane.showInputDialog("Escribe el nombre de la película que deseas alquilar");

                if (nombrePelicula.equalsIgnoreCase(pelicula) & disponibilidadP1) {
                    disponibilidadP1 = false;
                    JOptionPane.showMessageDialog(null, "Selección exitosa");
                } else if (nombrePelicula.equalsIgnoreCase(peliculaDos) & disponibilidadP2) {
                    disponibilidadP2 = false;
                    JOptionPane.showMessageDialog(null, "Selección exitosa");
                } else if (nombrePelicula.equalsIgnoreCase(pelicula) & disponibilidadP1 == false) {
                    JOptionPane.showMessageDialog(null, "Esa película no se encuentra disponible");
                } else if (nombrePelicula.equalsIgnoreCase(peliculaDos) & disponibilidadP2 == false) {
                    JOptionPane.showMessageDialog(null, "Esa película no se encuentra disponible");
                } else {
                    JOptionPane.showMessageDialog(null, "Esa película no hace parte de nuestra tienda");
                }
            }
            case 2 -> {
                String mensaje = "Película(s) disponoble(s):";
                if (disponibilidadP1) {
                    mensaje += "\n" + pelicula;
                }
                if (disponibilidadP2) {
                    mensaje += "\n" + peliculaDos;
                }
                if (mensaje.equals("Película(s) disponoble(s):")) {
                    mensaje = "No hay películas disponibles";
                }
                JOptionPane.showMessageDialog(null, mensaje);
            }
            case 3 -> {
                nombrePelicula = JOptionPane.showInputDialog("Escribe el nombre de la película que vas a recibir");
                if (disponibilidadP1 & disponibilidadP2) {
                    JOptionPane.showMessageDialog(null, "Esa película ya se encuentra en nuestra tienda");
                } else if (nombrePelicula.equalsIgnoreCase(pelicula)) {
                    disponibilidadP1 = true;
                    observacion = JOptionPane.showInputDialog("Observaciones");
                } else if (nombrePelicula.equalsIgnoreCase(peliculaDos)) {
                    disponibilidadP2 = true;
                    observacionDos = JOptionPane.showInputDialog("Observaciones");
                } else {
                    JOptionPane.showMessageDialog(null, "Esa película no hace parte de nuestra tienda");
                }
            }
            default -> JOptionPane.showMessageDialog(null, "La opción elegida no existe");
        }
    }

    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        Cuatro c = new Cuatro();
        c.agregarDatos();
        c.mostrar();
    }
}

La Droguería Mi Salud presta sus servicios en la localidad de Suba y requiere una aplicación para poder facturar los productos que vende a sus clientes y para ello, los productos tienen unas características que deben indicársele al cliente para que pueda escoger el producto a comprar. Para cada cliente, se tienen las opciones de compra de producto, consulta de precios por producto y devoluciones en caso de que se presenten.


package pseint.taller.dos;

import javax.swing.JOptionPane;

/**
 *
 * @author Juan Osorio
 */
public class Cinco {

    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        String mensaje, producto, productoDos;
        float precioParacetamol, PrecioIbuprofeno;
        int opcion, cantidad, cantidadParacetamol, cantidadIbuprofeno;

        producto = "paracetamol";
        productoDos = "ibuprofeno";
        precioParacetamol = 2000;
        PrecioIbuprofeno = 3000;
        cantidadParacetamol = 10;
        cantidadIbuprofeno = 15;

        opcion = Integer.parseInt(JOptionPane.showInputDialog("Droguería Mi Salud\n\nElija una opción:\n1. Comprar Producto\n2. Consultar producto\n3. Devolver producto"));

        switch (opcion) {
            case 1 -> {
                mensaje = JOptionPane.showInputDialog("Digite el nombre del producto que quiere comprar\n1. Paracetamol, Cantidad: " + cantidadParacetamol + "\n2. Ibuprofeno, Cantidad: " + cantidadIbuprofeno);
                if (mensaje.equalsIgnoreCase(producto) == false & mensaje.equalsIgnoreCase(productoDos) == false) {
                    JOptionPane.showMessageDialog(null, "Ese producto no se encuentra disponible en nuestra droguería");
                } else {
                    cantidad = Integer.parseInt(JOptionPane.showInputDialog("Digite la cantidad que quiere comprar"));
                    if (mensaje.equalsIgnoreCase(producto)) {
                        if (cantidad > 0 & cantidad <= cantidadParacetamol) {
                            cantidadParacetamol -= cantidad;
                            JOptionPane.showMessageDialog(null, "Su compra ha sido exitosa\nLas unidades restantes de este producto son: " + cantidadParacetamol);
                        } else {
                            JOptionPane.showMessageDialog(null, "La cantidad no puede ser inferior a 1 ni superior a " + cantidadParacetamol);
                        }
                    }
                    if (mensaje.equalsIgnoreCase(productoDos)) {
                        if (cantidad > 0 & cantidad <= cantidadParacetamol) {
                            cantidadIbuprofeno -= cantidad;
                            JOptionPane.showMessageDialog(null, "Su compra ha sido exitosa\nLas unidades restantes de este producto son: " + cantidadIbuprofeno);
                        } else {
                            JOptionPane.showMessageDialog(null, "La cantidad no puede ser inferior a 1 ni superior a " + cantidadIbuprofeno);
                        }
                    }
                }
            }
            case 2 -> {
                mensaje = JOptionPane.showInputDialog("Digite el nombre producto del cual quiere consultar su precio:\nParacetamol\nIbuprofeno");
                if (mensaje.equalsIgnoreCase(producto)) {
                    JOptionPane.showMessageDialog(null, "El precio del " + producto + " es de: $" + precioParacetamol);
                } else if (mensaje.equalsIgnoreCase(productoDos)) {
                    JOptionPane.showMessageDialog(null, "El precio del " + productoDos + " es de: $" + PrecioIbuprofeno);
                } else {
                    JOptionPane.showMessageDialog(null, "Ese producto no se encuentra disponible en nuestra droguería");
                }
            }
            case 3 -> {
                mensaje = JOptionPane.showInputDialog("Digite el producto que quiere devolver:\nParacetamol\nIbuprofeno");
                if (mensaje.equalsIgnoreCase(producto)) {
                    cantidad = Integer.parseInt(JOptionPane.showInputDialog("Digite la cantidad que quiere devolver"));
                    cantidadParacetamol += cantidad;
                    JOptionPane.showMessageDialog(null, "Devolución exitosa\nUnidaes disponibles de " + producto + ": " + cantidadParacetamol);
                } else if (mensaje.equalsIgnoreCase(productoDos)) {
                    cantidad = Integer.parseInt(JOptionPane.showInputDialog("Digite la cantidad que quiere devolver"));
                    cantidadIbuprofeno += cantidad;
                    JOptionPane.showMessageDialog(null, "Devolución exitosa\nUnidaes disponibles de " + productoDos + ": " + cantidadIbuprofeno);
                } else {
                    JOptionPane.showMessageDialog(null, "Ese producto no se encuentra disponible en nuestra droguería");
                }
            }
            default -> JOptionPane.showMessageDialog(null, "La opción elegida no existe");
        }
    }
}

7. La Secretaría de Salud Municipal requiere de una aplicación que le permita calcular el IMC (Índice de masa corporal) y requiere los datos peso en kilogramos y estatura en metros Para cada persona encuestada adicional a los datos suministrados, debe mostrar el resultado para cada uno y establecer en qué rango se encuentra (bajo peso, normal, sobrepeso y obeso).


package pseint.taller.dos;

import javax.swing.JOptionPane;

/**
 *
 * @author Juan Osorio
 */
public class Siete {

    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        double peso, estatura, imc;
        peso = Integer.parseInt(JOptionPane.showInputDialog("Secretaría de Salud Municipal\nPara calcular el índice de masa corporal (IMC), digite los siguientes datos:\nDigite su peso:"));
        if (peso >= 38.4 & peso <= 600) {
            estatura = Float.parseFloat(JOptionPane.showInputDialog("Digite su estatura:"));
            if (estatura >= 1.44 & estatura <= 2.72) {
                imc = peso / Math.pow(estatura, 2);
                if (imc < 18.5) {
                    JOptionPane.showMessageDialog(null, "Su IMC es " + imc + " lo que indica que su peso está en la categoría:\nBajo peso");
                }
                if (imc >= 18.5 & imc <= 24.9) {
                    JOptionPane.showMessageDialog(null, "Su IMC es " + imc + " lo que indica que su peso está en la categoría:\nNormal");
                }
                if (imc >= 25 & imc <= 29.9) {
                    JOptionPane.showMessageDialog(null, "Su IMC es " + imc + " lo que indica que su peso está en la categoría:\nSobrepeso");
                }
                if (imc >= 30) {
                    JOptionPane.showMessageDialog(null, "Su IMC es " + imc + " lo que indica que su peso está en la categoría:\nObesidad");
                }
            } else {
                JOptionPane.showMessageDialog(null, "Su estatura no puede ser inferior a 1.44 m ni superior a 2.72 m");
            }
        } else {
            JOptionPane.showMessageDialog(null, "Su peso no puede ser inferior a 38.4 kg ni superior a 600 kg");
        }
    }
}

El profesor de geometría está explicando a sus estudiantes las fórmulas para calcular el área de diferentes figuras geométricas, para ello requiere una aplicación que le facilite el ejercicio solicitándole los valores al estudiante. La aplicación debe permitir que el estudiante seleccione si desea calcular el área de un rectángulo, triángulo o trapecio. No olvide solicitar los datos necesarios para realizar cada cálculo y mostrar su respectivo resultado.

package pseint.taller.dos;

import javax.swing.JOptionPane;

/**
 *
 * @author Juan Osorio
 */
public class Nueve {

    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        float x, y, z, resultado;

        int opcion = Integer.parseInt(JOptionPane.showInputDialog("Escoja le figura geómetrica de la que desea calcular el área\n1. Rectángulo\n2. Triángulo\n3. Trapecio"));

        switch (opcion) {
            case 1 -> {
                x = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor del largo del rectángulo"));
                y = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor del ancho del rectángulo"));
                resultado = x * y;
                JOptionPane.showMessageDialog(null, "El valor del área del rectángulo es de: " + resultado);
            }
            case 2 -> {
                x = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor de la base del triángulo"));
                y = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor de la altura del triángulo"));
                resultado = x * y / 2;
                JOptionPane.showMessageDialog(null, "El valor del área del triángulo es de: " + resultado);
            }
            case 3 -> {
                x = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor de la base mayor del trapecio"));
                y = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor de la base menor del trapecio"));
                z = Float.parseFloat(JOptionPane.showInputDialog("Digite el valor de la altura del trapecio"));
                resultado = ((x + y) * z) / 2;
                JOptionPane.showMessageDialog(null, "El valor del área del trapecio es de: " + resultado);
            }
            default -> JOptionPane.showMessageDialog(null, "La opción elegida no existe");
        }
    }
}

El banco "Su banco fiel" es un banco que inicia sus actividades financieras y necesita una aplicación para llevar las cuentas de sus usuarios; por lo tanto, se sugiere que la cuenta tenga los atributos titular y cantidad. Para cada cliente las cuentas permitirán realizar ingresos, retiros o consultas de valor. En los ingresos no se pueden insertar valores negativos y para los retiros el valor no puede ser mayor al valor que tiene en la cuenta.



package pseint.taller.dos;

import javax.swing.JOptionPane;

/**
 *
 * @author Juan Osorio
 */
public class Diez {

    /**
     * @param args the command line arguments
     */
    public static void main(String args[]) {
        String titular, mensaje;
        float saldo, dinero;
        int opcion;

        titular = "Juan Osorio";
        saldo = 1000;

        mensaje = JOptionPane.showInputDialog("Su Banco Fiel\n\nDigite el nombre del titular");
        if (mensaje.equalsIgnoreCase(titular)) {
            opcion = Integer.parseInt(JOptionPane.showInputDialog("Titular de la cuenta: " + titular + "\nNúmero de la cuenta: 7481950263\nTipo de cuenta: Ahorros\n\nElija una opción:\n1. Realizar ingreso\n2. Realizar retiro\n3. Consulta"));

            switch (opcion) {
                case 1 -> {
                    dinero = Float.parseFloat(JOptionPane.showInputDialog("Digite la cantidad de dinero que desea ingresar"));
                    if (dinero > 0) {
                        saldo = saldo + dinero;
                        JOptionPane.showMessageDialog(null, "Su saldo actual quedó en: $" + saldo);
                    } else {
                        JOptionPane.showMessageDialog(null, "La cantidad de dinero a ingresar no puede ser 0 o inferior");
                    }
                }
                case 2 -> {
                    dinero = Float.parseFloat(JOptionPane.showInputDialog("Digite la cantidad de dinero que desea retirar"));
                    if (dinero <= saldo) {
                        saldo = saldo - dinero;
                        JOptionPane.showMessageDialog(null, "Su saldo actual quedó en: $" + saldo);
                    } else {
                        JOptionPane.showMessageDialog(null, "La cantidad de dinero a retirar no puede superar la cantidad de dinero disponible");
                    }
                }
                case 3 -> JOptionPane.showMessageDialog(null, "El saldo que tiene actualmente es de: $" + saldo);
                default -> JOptionPane.showMessageDialog(null, "La opción elegida no existe");
            }
        } else {
            JOptionPane.showMessageDialog(null, "No hay ninguna cuenta asociada a ese nombre");
        }
    }
}
